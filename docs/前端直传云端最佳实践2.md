在 React 19 + 结合 React Query 5 的现代工程中，我们将**完全剥离 UI 与状态/网络层**。

React Query 5 非常适合处理这种异步副作用，但它本身不包含“进度条（Progress）”的概念。因此，最佳实践是：**用 React Query 5 管理请求生命周期（isPending, isSuccess, error），用 React 本地状态（useState）管理实时进度。**

为了实现这套基于环境变量（开发环境阿里云，其他环境 S3）的无缝切换，下面是为你量身定制的完整落地指南。

#### 2. 阿里云适配器 (`src/lib/upload/ali-adapter.ts`)

```typescript
import OSS from 'ali-oss';
import type { STSData, UploadAdapter, UploadResult } from './types';

export class AliOSSAdapter implements UploadAdapter {
  constructor(private sts: STSData) {}

  async upload(file: File, onProgress?: (percent: number) => void): Promise<UploadResult> {
    const client = new OSS({
      region: this.sts.region,
      accessKeyId: this.sts.accessKeyId,
      accessKeySecret: this.sts.accessKeySecret,
      stsToken: this.sts.securityToken,
      bucket: this.sts.bucket,
      // 开启 HTTPS
      secure: true, 
    });

    const extension = file.name.split('.').pop();
    const objectKey = `uploads/${Date.now()}-${Math.random().toString(36).slice(2)}.${extension}`;

    // multipartUpload 自动处理大于 100KB 的文件
    const result = await client.multipartUpload(objectKey, file, {
      progress: (p) => {
        if (onProgress) onProgress(Math.floor(p * 100));
      },
      // 并发上传的分片数量
      parallel: 4, 
      // 分片大小，推荐 1MB - 5MB 之间
      partSize: 1024 * 1024 * 2, 
    });

    // 根据你的 Bucket 配置生成访问 URL
    const url = result.res.requestUrls[0].split('?')[0]; 
    
    return { url, key: result.name };
  }
}

```

#### 3. Amazon S3 适配器 (`src/lib/upload/s3-adapter.ts`)

这里使用 `@aws-sdk/lib-storage`，它是 AWS 官方推荐处理大文件分片的现代方案。

```typescript
import { S3Client } from '@aws-sdk/client-s3';
import { Upload } from '@aws-sdk/lib-storage';
import type { STSData, UploadAdapter, UploadResult } from './types';

export class AWSS3Adapter implements UploadAdapter {
  constructor(private sts: STSData) {}

  async upload(file: File, onProgress?: (percent: number) => void): Promise<UploadResult> {
    const client = new S3Client({
      region: this.sts.region,
      credentials: {
        accessKeyId: this.sts.accessKeyId,
        secretAccessKey: this.sts.accessKeySecret,
        sessionToken: this.sts.securityToken,
      },
    });

    const extension = file.name.split('.').pop();
    const objectKey = `uploads/${Date.now()}-${Math.random().toString(36).slice(2)}.${extension}`;

    const parallelUploads3 = new Upload({
      client,
      params: { 
        Bucket: this.sts.bucket, 
        Key: objectKey, 
        Body: file,
        ContentType: file.type // 确保在 S3 中预览正常
      },
      queueSize: 4, 
      partSize: 1024 * 1024 * 5, // AWS 最小分片要求 5MB
    });

    parallelUploads3.on('httpUploadProgress', (progress) => {
      if (onProgress && progress.loaded && progress.total) {
        onProgress(Math.floor((progress.loaded / progress.total) * 100));
      }
    });

    await parallelUploads3.done();
    
    const url = `https://${this.sts.bucket}.s3.${this.sts.region}.amazonaws.com/${objectKey}`;
    return { url, key: objectKey };
  }
}

```

#### 4. 结合 React Query 5 的 Hook (`src/hooks/useCloudUpload.ts`)

这是连接业务层和底层 SDK 的核心枢纽。我们通过环境变量在这里进行拦截分配。

```typescript
import { useState } from 'react';
import { useMutation } from '@tanstack/react-query';
import { AliOSSAdapter } from '../lib/upload/ali-adapter';
import { AWSS3Adapter } from '../lib/upload/s3-adapter';
import type { UploadResult, STSData } from '../lib/upload/types';

// 模拟后端接口请求
const fetchSTSFromBackend = async (): Promise<STSData> => {
  const res = await fetch('/api/get-upload-token');
  if (!res.ok) throw new Error('Failed to fetch STS token');
  return res.json();
};

export const useCloudUpload = () => {
  const [progress, setProgress] = useState(0);

  // 严格的 React Query v5 语法
  const mutation = useMutation<UploadResult, Error, File>({
    mutationFn: async (file: File) => {
      setProgress(0);
      
      // 1. 获取凭证
      const sts = await fetchSTSFromBackend();

      // 2. 根据 Vite 环境变量实例化具体的 SDK
      const AdapterClass = import.meta.env.MODE === 'development' 
        ? AliOSSAdapter 
        : AWSS3Adapter;
        
      const adapter = new AdapterClass(sts);

      // 3. 执行上传并监听进度
      return await adapter.upload(file, (p) => setProgress(p));
    },
    onSuccess: (data) => {
      // 可以在这里统一处理上传成功后的 Toast 提示
      console.log('Upload success, URL:', data.url);
    },
    onError: (error) => {
      setProgress(0);
      console.error('Upload failed:', error);
    }
  });

  return {
    upload: mutation.mutateAsync, // 暴露 mutateAsync 方便外层 await
    isUploading: mutation.isPending, // v5 中 isLoading 已被替换为 isPending
    progress,
    error: mutation.error,
    isSuccess: mutation.isSuccess
  };
};

```

#### 5. Tailwind CSS 组件 (`src/components/Uploader.tsx`)

现在，你的 UI 组件将变得极其干净，只负责渲染和事件绑定。

```tsx
import { useCallback } from 'react';
import { useCloudUpload } from '../hooks/useCloudUpload';

export const Uploader = () => {
  const { upload, isUploading, progress, error, isSuccess } = useCloudUpload();

  const handleFileChange = useCallback(async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file) return;

    // 前端基础校验
    if (file.size > 1024 * 1024 * 500) {
      alert('文件不能超过 500MB');
      return;
    }

    try {
      // 将返回的 URL 发送给业务后端进行数据绑定
      const result = await upload(file);
      await fetch('/api/business/save', {
        method: 'POST',
        body: JSON.stringify({ attachmentUrl: result.url })
      });
    } catch (err) {
      // 错误已在 Hook 中通过 error 状态暴露，这里可选做额外处理
    }
    
    // 清空 input，允许重复上传同一个文件
    e.target.value = '';
  }, [upload]);

  return (
    <div className="flex flex-col items-center justify-center w-full max-w-md p-6 border-2 border-dashed rounded-lg border-gray-300 bg-gray-50 hover:bg-gray-100 transition-colors relative overflow-hidden">
      
      {/* 进度条背景 */}
      {isUploading && (
        <div 
          className="absolute left-0 top-0 h-full bg-blue-100 opacity-50 transition-all duration-300 ease-out z-0"
          style={{ width: `${progress}%` }}
        />
      )}

      <div className="relative z-10 flex flex-col items-center">
        <svg className="w-10 h-10 text-gray-400 mb-3" fill="none" stroke="currentColor" viewBox="0 0 24 24">
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M7 16a4 4 0 01-.88-7.903A5 5 0 1115.9 6L16 6a5 5 0 011 9.9M15 13l-3-3m0 0l-3 3m3-3v12" />
        </svg>
        <p className="text-sm text-gray-600 font-medium">
          {isUploading ? `上传中... ${progress}%` : '点击或拖拽文件上传'}
        </p>
        
        {error && <p className="text-xs text-red-500 mt-2">{error.message}</p>}
        {isSuccess && !isUploading && <p className="text-xs text-green-500 mt-2">上传成功！</p>}
      </div>

      <input 
        type="file" 
        className="absolute inset-0 w-full h-full opacity-0 cursor-pointer z-20"
        onChange={handleFileChange}
        disabled={isUploading}
      />
    </div>
  );
};

```

### 四、 性能优化与避坑指南

在这套架构落地的过程中，有几个经常踩坑的点需要注意：

1. **STS 凭证缓存：** 每次上传都去后端请求 STS 是比较消耗性能的（特别是当网络协议如你之前研究的从 HTTP/1.1 切换中产生握手延迟时）。STS 通常有 15-60 分钟的有效期。最佳实践是在 `fetchSTSFromBackend` 中做一个内存缓存，校验没过期就直接用本地缓存的凭证。
2. **CORS 跨域问题：** 这是联调时 90% 的报错来源。必须要让后端在阿里云和 S3 的控制台配置 CORS 规则：
* `AllowedOrigins`: 配置为你本地和线上的域名。
* `AllowedMethods`: `GET, PUT, POST, DELETE, HEAD`。
* `AllowedHeaders`: `*`。
* `ExposeHeaders`: `ETag` (分片合并时需要验证)。


3. **分片大小（PartSize）设置：** 阿里云的分片最小是 100KB，但 **AWS S3 强制要求除最后一个分片外，其他分片不能小于 5MB**。这也是为什么我在上面的适配器代码中，将 AWS 的 `partSize` 设置为 `1024 * 1024 * 5` 的原因。
